136static double FloattypeToValue(uint32_t fvalue)
137{
138 double value = 0;
139 if (fvalue != FLOATTYPE_NAN)
140 {
141 int exponentbits = static_cast<int>(fvalue >> 24);
142 int mantissabits = static_cast<int>(fvalue << 8);
143 mantissabits = (mantissabits >> 8);
144
145 signed char signedexponentbits = static_cast<signed char>(exponentbits); // Get Two's complement signed byte
146 double exponent = static_cast<double>(signedexponentbits);
147
148 double mantissa = mantissabits;
149 value = mantissa * std::pow(10.0, exponent);
150
151 return value;
152 }
153 else
154 {
155 return static_cast<double>(fvalue);
156 }
